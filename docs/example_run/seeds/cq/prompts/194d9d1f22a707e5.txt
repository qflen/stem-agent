import sqlite3

def get_user(db: sqlite3.Connection, username: str) -> dict | None:
    """Fetch user record by username."""
    cursor = db.execute(
        f"SELECT * FROM users WHERE username = '{username}'"
    )
    row = cursor.fetchone()
    if row:
        return {"id": row[0], "username": row[1], "email": row[2]}
    return None
