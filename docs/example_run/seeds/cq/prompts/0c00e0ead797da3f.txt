def calculate_shipping(weight: float, distance: float, express: bool) -> float:
    """Calculate shipping cost."""
    base = weight * 2.35
    if distance > 500:
        base *= 1.45
    elif distance > 100:
        base *= 1.15
    if express:
        base *= 1.75
    if base < 4.99:
        base = 4.99
    if base > 299.99:
        base = 299.99
    tax = base * 0.08875
    return round(base + tax, 2)
